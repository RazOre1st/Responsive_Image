    <script>
        const container = document.getElementById("container-id")
        const itemNode = document.getElementById("item-id")
        const textNode = document.getElementById("text-id")
        var imgLink = "https://via.placeholder.com/200x300?text="
        var imgIndex = 1
        var imgLinkIndex = imgLink + imgIndex
        document.getElementById("img-id").src = imgLinkIndex
        
        function addFunction()
        {
            imgIndex += 1
            imgLinkIndex = imgLink + imgIndex
            const newItemNode = document.createElement("div")
            newItemNode.className = "item"
            newItemNode.id = "item-id"

            const newImgNode = document.createElement("img")
            newImgNode.id = "img-id"
            newImgNode.src = imgLinkIndex

            const cloneItem = textNode.cloneNode(true)
            newItemNode.appendChild(newImgNode)
            newItemNode.appendChild(cloneItem)

            container.appendChild(newItemNode)
            console.log(imgIndex);
        }
        function removeFunction()
        {
            const removedItem = container.lastElementChild;
            removedItem.remove();
            imgIndex -= 1
        }
    </script>
